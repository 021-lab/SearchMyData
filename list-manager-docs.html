Структура данных

list-data.js — плоский массив

let items = [
  { id: 2,  line1: 'Хлеб ржаной' },
  { id: 26, line1: 'Бородинский', line2: '400 г', level: 1 },
  { id: 27, line1: 'Столичный',   line2: '500 г', level: 1 },
];
let nextId = 32; // монотонно растущий, ID не переиспользуются
ПолеТипОписание
id number Уникальный идентификатор элемента
line1 string Заголовок (обязательно)
line2 string ? Подпись / дополнительный текст
level number ? Глубина вложенности: 0 = корень (по умолчанию), 1 = дочерний и т.д.
collapsed boolean ? Подсписок свёрнут. Управляется через toggleCollapse()
tags string[] ? Список тегов, назначенных элементу через левую панель

Вложенность и сворачивание

render() — линейный проход по плоскому массиву

let collapsedAt = Infinity;

items.forEach((item, idx) => {
  const level  = item.level || 0;
  const hidden = level > collapsedAt;   // глубже свёрнутого родителя

  if (!hidden) collapsedAt = item.collapsed ? level : Infinity;

  const hasChildren = (items[idx + 1]?.level || 0) > level;

  wrapper.style.display    = hidden ? 'none' : '';
  wrapper.style.marginLeft = (level * 24) + 'px';
});
Дерево не строится — level читается прямо из элемента. Отступ = level × 24 px.

toggleCollapse() — пишет флаг и вызывает render()

function toggleCollapse(itemId) {
  const item = items.find(i => i.id === itemId);
  const hasChildren = /* следующий элемент глубже */;
  if (!hasChildren) return;
  item.collapsed = !item.collapsed;
  render();
}

Изменение данных

ДействиеМутация items[]
Добавить items.push({ id: nextId++, ... })
Добавить вложенный items.splice(insertIdx, 0, { ..., level: parent.level + 1 })
Редактировать item.line1 = ...; item.line2 = ...
Удалить items = items.filter(i => i.id !== itemId)
Перетащить После drop DOM-порядок враппе­ров читается обратно:
items = [...wrappers].map(w => items.find(it => it.id === +w.dataset.id))
Тег item.tags.push(tag) или item.tags.splice(idx, 1)
Отмена (undo) items = undoSnapshot — глубокая копия, сохранённая перед drag
Все данные живут только в памяти. Перезагрузка страницы возвращает состояние из list-data.js.
